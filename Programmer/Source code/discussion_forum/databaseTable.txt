CREATE TABLE `lmhs8tez_idea_forum`.`users` ( `id` INT(50) NOT NULL AUTO_INCREMENT , `email` VARCHAR(150) NOT NULL , `password` VARCHAR(150) NOT NULL , `user_role` TINYINT(4) NOT NULL DEFAULT '5' , `user_status` TINYINT(5) NOT NULL DEFAULT '1' , `remember_token` VARCHAR(100) NOT NULL , `created_at` TIMESTAMP NULL , `updated_at` TIMESTAMP NULL , PRIMARY KEY (`id`), UNIQUE (`email`)) ENGINE = InnoDB;
